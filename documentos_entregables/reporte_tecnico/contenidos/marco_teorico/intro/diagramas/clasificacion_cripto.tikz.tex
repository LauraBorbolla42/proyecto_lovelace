%
% Diagrama de clasificación de la criptografía,
% Capítulo de marco teórico.
% Proyecto Lovelace.
%

\begin{tikzpicture}[
  clasificador/.style = {
    rectangle,
    draw = black,
    thin,
    inner sep = 4mm,
    text width = 25mm,
    align = center,
    minimum height = 1.5cm}
  ]

  % Clasificadores

  \node[clasificador]
    (transposicion)
    {\small Transposición};
  \node[clasificador]
    (sustitucion)
    [right = of transposicion]
    {\small Sustitución};
  \node[clasificador]
    (simetrica)
    [right = of sustitucion]
    {\small Simétrica};
  \node[clasificador]
    (asimetrica)
    [right = of simetrica]
    {\small Asimétrica};

  \node[clasificador]
    (clasica)
    at ($(transposicion)!0.5!(sustitucion) + (0, 1.0in)$)
    {\small Clásica};
  \node[clasificador]
    at ($(simetrica)!0.5!(asimetrica) + (0, 1.0in)$)
    (moderna)
    {\small Moderna};

  \node[clasificador]
    at ($(clasica)!0.5!(moderna) + (0, 1.0in)$)
    (criptografia)
    {\small Criptgrafía};

  % Relaciones
  \draw[-{Triangle[open]}]
    (transposicion.north)
    |-
    ($(clasica) - (0in, 0.5in)$)
    --
    (clasica.south);
  \draw[-{Triangle[open]}]
    (sustitucion.north)
    |-
    ($(clasica) - (0in, 0.5in)$)
    --
    (clasica.south);

  \draw[-{Triangle[open]}]
    (simetrica.north)
    |-
    ($(moderna) - (0in, 0.5in)$)
    --
    (moderna.south);
  \draw[-{Triangle[open]}]
    (asimetrica.north)
    |-
    ($(moderna) - (0in, 0.5in)$)
    --
    (moderna.south);

  \draw[-{Triangle[open]}]
    (clasica.north)
    |-
    ($(criptografia) - (0in, 0.5in)$)
    --
    (criptografia.south);
  \draw[-{Triangle[open]}]
    (moderna.north)
    |-
    ($(criptografia) - (0in, 0.5in)$)
    --
    (criptografia.south);

\end{tikzpicture}
